# This is the easiest way - Railway provides this automatically
# Just reference it from your database service:
MYSQL_URL=${{YourDatabaseServiceName.MYSQL_URL}}

# Database Configuration

# Application Configuration
PORT=3005

# CORS Configuration (Update with your actual Railway domain)
ALLOWED_ORIGINS=https://your-app-name.up.railway.app

# IMPORTANT NOTES:
# 1. If using MYSQL_URL, you don't need DB_HOST, DB_PORT, DB_NAME, DB_USER, DB_PASSWORD
